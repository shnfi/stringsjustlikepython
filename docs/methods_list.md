capitalize(char *str) // Converts the first character to upper case
casefold(char *str) // Converts string into lower case
center(char *str, unsigned int width) // Returns a centered string
endswith(char *str, char last_char) // Returns true if the string ends with the specified value
find(char *str, char char_) // Searches the string for a specified value and returns the position of where it was found
isalpha(char *str) // Returns True if all characters in the string are in the alphabet
isdecimal(char *str) // Returns True if all characters in the string are decimals
isidentifier(char *str) // Returns True if the string is an identifier
isallnum(char *str) // Returns True if all characters in the string are alphanumeric
islower(char *str) // Returns True if all characters in the string are lower case
